def rotz(theta):
    cs = np.cos(theta)
    sn = np.sin(theta)
    R  = np.array([[cs, sn, 0],[-sn, cs, 0], [0, 0, 1]])
    R=np.around(R,decimals=15)+0
    return R
\end{python}
